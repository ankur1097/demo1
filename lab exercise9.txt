class bank:
    def name(self,name):
        self.name=name
    def acc(self,acc):
        self.acc=acc
    def typ(self,typ):
        self.typ=typ
    def dep(self,dep):
        self.dep=dep
    def withdr(self,withdr):
        self.withdr=withdr
        self.remain=self.dep-self.withdr
    def display(self):
        print("customer name:",self.name)
        print("account number:",self.acc)
        print("account type:",self.typ)
        print("balance amount:",self.remain)
m=bank()

while True:
    print("enter your choice:")
    print("""1. New customer
2. Deposit
3. Withdrawal
4. Display
5. Exit""")
    a=int(input())
    
    if a==1:
        name=input("enter customer name:")
        number=int(input("enter account number:"))
        typ=input("enter account type:")
        m.name(name)
        m.acc(number)
        m.typ(typ)
    if a==2:
       k=int(input("enter deposit amount:"))
       m.dep(k)
        
    if a==3:
        p=int(input("enter withdrawal amount:"))
        m.withdr(p)
    if a==4:
        m.display()
    if a==5:
        break
